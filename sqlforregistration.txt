CREATE DATABASE IF NOT EXISTS student_registration;
USE student_registration;

CREATE TABLE IF NOT EXISTS students (
  id INT AUTO_INCREMENT PRIMARY KEY,
  ht_no VARCHAR(50),
  student_name VARCHAR(255),
  branch VARCHAR(100),
  dob DATE,
  aadhar VARCHAR(20),
  mobile VARCHAR(20),
  email VARCHAR(255),
  ssc_marks VARCHAR(50),
  ssc_school VARCHAR(255),
  inter_marks VARCHAR(50),
  inter_college VARCHAR(255),
  eamcet_htno VARCHAR(50),
  eamcet_year VARCHAR(10),
  eamcet_rank VARCHAR(20),
  ecet_htno VARCHAR(50),
  ecet_year VARCHAR(10),
  ecet_rank VARCHAR(20),
  cat_eamcet TINYINT(1) DEFAULT 0,
  cat_spot TINYINT(1) DEFAULT 0,
  cat_ecet TINYINT(1) DEFAULT 0,
  nationality VARCHAR(100),
  caste VARCHAR(100),
  state VARCHAR(50),
  fee VARCHAR(10),
  father_qualification VARCHAR(255),
  father_occupation VARCHAR(255),
  father_income VARCHAR(50),
  father_landline VARCHAR(50),
  father_mobile VARCHAR(20),
  father_email VARCHAR(255),
  mother_qualification VARCHAR(255),
  mother_occupation VARCHAR(255),
  mother_income VARCHAR(50),
  mother_landline VARCHAR(50),
  mother_mobile VARCHAR(20),
  mother_email VARCHAR(255),
  permanent_address TEXT,
  contact_address TEXT,
  hostel_type VARCHAR(255),
  hostel_name VARCHAR(255),
  hostel_room VARCHAR(50),
  guardian_name VARCHAR(255),
  guardian_address VARCHAR(255),
  guardian_phone VARCHAR(20),
  guardian_mobile VARCHAR(20),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
